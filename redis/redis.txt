## Redis Deployment (Production Grade)

helm repo add bitnami https://charts.bitnami.com/bitnami
helm repo update

helm install redis bitnami/redis \
  --namespace cache \
  --create-namespace \
  --set architecture=replication \
  --set auth.enabled=true \
  --set auth.password=StrongPassword123 \
  --set master.persistence.enabled=true \
  --set replica.replicaCount=2 \
  --set replica.persistence.enabled=true


  ## vi values.yml

  service:
  type: ClusterIP

  ##  helm upgrade redis bitnami/redis -f values.yml -n cache

  ## vi pom.xml

  <dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-data-redis</artifactId>
</dependency>

  <dependency>
    <groupId>org.springframework.boot</groupId>
    <artifactId>spring-boot-starter-data-cache</artifactId>
</dependency>

## src/application.yml

spring:
  data:
    mongodb:
      host: ${MONGO_DB_HOSTNAME}
      port: 27017
      username: ${MONGO_DB_USERNAME}
      password: ${MONGO_DB_PASSWORD}
      database: users
      authentication-database: admin

  redis:
    host: redis-master.cache.svc.cluster.local
    port: 6379
    password: ${REDIS_PASSWORD}
    timeout: 6000

  cache:
    type: redis
    redis:
      time-to-live: 600000   # 10 minutes

## kubectl create secret generic redis-secret --from-literal=password=StrongPassword123 -n prod


## vi spring.yaml 

env:
